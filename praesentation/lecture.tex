        \end{definition}
        \onslide<2->{
        \begin{example}{}

        \end{example}
        }
    \mynextcolumn
    \vspace{-10mm}
        \onslide<3->{
        \begin{notetight}{}
            \centering
            \includegraphics[width=\linewidth]{code/#3.pdf}
        \end{notetight}
        }
    \end{mycolumns}
}
